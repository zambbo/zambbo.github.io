#include<iostream>


int DP(int n) {
    int* dp = new int[n + 1];
    for (int i = 1; i <= n; i++)
        dp[i] = 0;
    
    dp[0] = 1;
    for (int i = 1; i <= 3; i++) {
        for (int j = i;j<=n;j++)
            dp[j] += dp[j - i];
    }
    
    return dp[n];
}


int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(NULL);
    std::cout.tie(NULL);


    int T; // num Of TestCase
